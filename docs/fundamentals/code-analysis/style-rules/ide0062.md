### <a name="example"></a><span data-ttu-id="c7c44-127">Ejemplo</span><span class="sxs-lookup"><span data-stu-id="c7c44-127">Example</span></span>

    {

    }
}

{
    Hello();
    void Hello()
    {
        Console.WriteLine("Hello");
    }
}
